```ts:中文匹配
  const SYMBOL_REGEX = /[\u3002\uff1b\uff0c\uff1a\u201c\u201d\uff08\uff09\u3001\uff1f\u300a\u300b]/;
  const WORD_REGEX = /[\u3400-\u9FBF]/;

  function hasChinese(value:any) {
    return WORD_REGEX.test(value) || SYMBOL_REGEX.test(value);
  }
```
