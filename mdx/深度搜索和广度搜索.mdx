        },
      ],
    },
  ]

  function findName(id, arr) {
    if (!Array.isArray(arr)) return '';
    for (let i = 0; i < arr.length; i++) {
      let item = arr[i];
      if (item.id == id) return item.name;
      let str = findName(id, item.children);
      if (str) return str;
    }
  }
